\section{Overview}

\subsection{Objectives and competences}

\begin{frame}[t]{Course}
\begin{itemize}
  \item \textmark{Goal}: to achieve that students know the basic concepts related
to the \textemph{computer architecture} and the impact that these concepts have on the
\textgood{performance of applications} and \textgood{computer systems}.

\end{itemize}
\end{frame}

\begin{frame}[t]{Competences and skills}
\begin{itemize}
  \item To achieve this \textgood{goal}, the student will gain insight in the following \textmark{skills}:
    \begin{itemize}
      \mode<presentation>{\vfill\pause}
      \item Ability to know, understand and evaluate 
            \textgood{computer architecture}, 
            as well as its \textmark{basic components}.

      \mode<presentation>{\vfill\pause}
      \item Knowledge and application of 
            \textmark{fundamental principles} and \textmark{basic techniques} for 
            \textgood{concurrent and parallel programming}.

      \mode<presentation>{\vfill\pause}
      \item Ability to \textmark{analyze and evaluate} \textgood{computer architectures}, 
            including \textgood{parallel platforms}, 
            as well as to be able to develop and \textmark{optimize software} 
            for those architectures.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Previous skills}
\begin{itemize}
  \item We will be using skills and contents from:
    \begin{itemize}

      \mode<presentation>{\vfill}
      \item \textmark{Computer Structure}.
        \begin{itemize}
          \item Assembly programming.
          \item Processor.
          \item Memory hierarchy.
        \end{itemize}

      \mode<presentation>{\vfill}
      \item \textmark{Operating Systems}.
        \begin{itemize}
          \item C programming language.
          \item Processes and Threads.
          \item Multithreaded programming.
        \end{itemize}

      \mode<presentation>{\vfill}
      \item \textmark{Programming}.
        \begin{itemize}
          \item Programming techniques.
        \end{itemize}
    \end{itemize}

\end{itemize}
\end{frame}
