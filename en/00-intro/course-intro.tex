\section{Overview}

\subsection{Objectives and competences}

\begin{frame}[t]{Course}
\begin{itemize}
  \item \bulletgood{Goal}: to achieve that students know the basic concepts related
to the \bulletmark{computer architecture} and the impact that these concepts have on the
\bulletgood{performance of applications} and computer systems.

\end{itemize}
\end{frame}

\begin{frame}[t]{Competences and skills}
\begin{itemize}
  \item To achieve this \bulletgood{goal}, the student will gain insight in the following \bulletmark{skills}:
    \begin{itemize}
      \mode<presentation>{\vfill\pause}
      \item Ability to know, understand and evaluate \bulletmark{computer architecture}, as well as its \bulletmark{basic components}.

      \mode<presentation>{\vfill\pause}
      \item Knowledge and application of fundamental principles and basic techniques for \bulletmark{concurrent and parallel programming}.

      \mode<presentation>{\vfill\pause}
      \item Ability to analyze and evaluate computer architectures, including \bulletmark{parallel platforms}, as well as to be able to develop and optimize \bulletmark{software for those architectures}.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Previous skills}
\begin{itemize}
  \item We will be using skills and contents from:
    \begin{itemize}
      \item \bulletmark{Computer Structure}.
      \item \bulletmark{Operating Systems}.
      \item \bulletmark{Programming}.
    \end{itemize}

\end{itemize}
\end{frame}

\subsection{Program}

\begin{frame}[t]{Contents}
\begin{enumerate}
  \item Fundamentals of computer design.
  \item Performance evaluation in computer systems.
  \item Memory hierarchy.
  \item Instruction Level Parallelism.
  \item Introduction to multiprocessors.
  \item Parallel and concurrent programming models.
\end{enumerate}
\end{frame}
