\section{Assessment}

\begin{frame}[t]{Assessment System}
\begin{itemize}
  \item In summary:
    \mode<presentation>{\vfill}
    \begin{itemize}
      \item \bulletenum{Final exam}: 40\% of final score.
        \begin{itemize}
          \item All contents included (theoretical, practical, and projects).
        \end{itemize}

      \mode<presentation>{\vfill}
      \item \bulletenum{Continuous evaluation}: 60\% of final score.
        \begin{itemize}
          \item \bulletmark{Tests}: 25\% of final score.
          \item \bulletmark{Projects}: 35\% of final score.
        \end{itemize}

    \mode<presentation>{\vfill}
    \item Evaluation rounds:
      \begin{itemize}
        \item Ordinary round: January.
        \item Extraordinary round: June.
      \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Continuous evaluation}
\begin{itemize}
  \item Good continuous evaluation scores are \bulletmark{key} for success.

  \mode<presentation>{\vfill}
  \item \bulletenum{Elements}:
    \begin{itemize}
      \item \bulletmark{Midterm exam}: 25\% of final score.
      \item \bulletmark{Projects}: 35\% of final score.

  \vspace{1em}

  \mode<presentation>{\vfill}
  \item You \alert{have not followed} continuous evaluation:

      \item if you get less than 2.0 in a project/lab, or,
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Ordinary round: Continuous evaluation}
\begin{itemize}
  \item If you follow the continuous evaluation process:
    \begin{itemize}
    \item \bulletenum{Final exam}: 40\%.
      \begin{itemize}
        \item Minimum required: 3.5.
      \end{itemize}
    \item \bulletenum{Tests}: 25\%
      \begin{itemize}
        \item Minimum required: There is no minimum.
      \end{itemize}
    \item \bulletenum{Projects/Labs}: 35\%.
      \begin{itemize}
        \item Minimum required per project/lab: 2.0.
      \end{itemize}
    \item If you do not achieve a minimum, the average is not computed and \alert{you have failed}.
  \end{itemize}
  \item \bulletgood{Bonus}:
    \begin{itemize}
      \item 1 point is added to your score if:
        \begin{itemize}
          \item You got at least 7.0 points in continuous evaluation, and,
          \item You got at least 6.0 points in the final exam.
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Labs/Projects}
\begin{itemize}
  \item Four lab sessions.
    \begin{itemize}
      {\scriptsize
        \item Performed during lab sessions in groups of 2 students.
        \item Test or report after the session during a given period.
        \item Each lab: 3\% of total mark.
        \item Minumum mark: 2 over 10.
      }
    \end{itemize}
  \item One parallel programming project.
    \begin{itemize}
      {\scriptsize
        \item Developed using OpenMP and C++ language.
        \item Two deliveries.
        \begin{itemize}
          {\scriptsize
          \item Week 8 (10/24): Sequential optimizations.
          \item Week 11 (11/14): Parallel optimizations.
          }
        \end{itemize}

        \item Groups of 4 people.
        \item Performance is a key criterion.
        \item Marks include quality of code, tests, and performance evaluation.
        \item Importance of quality of project report.
        \item Weight: 23\% of total mark.
        \item Minimum mark: 2 over 10.
      }
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Extraordinary round}
\begin{itemize}
  \item Extraordinary exam to be held in June.
  \vspace{1em}
  \item Rules:
    \begin{enumerate}
      \item Student completed continuous evaluation:
        \begin{itemize}
          \item Extraordinary exam gives 40\% and continuous evaluation gives 60\%.
          \item Only applied if the score in exam is at least 3.5.
        \end{itemize}
      \item Student did not complete continuous evaluation: 
        \begin{itemize}
          \item Final exam gives 100\%.
        \end{itemize}
    \end{enumerate}
    \begin{itemize}
      \item For students having completed continuous evaluation best of two options is taken.
    \end{itemize}
\end{itemize}
\end{frame}
