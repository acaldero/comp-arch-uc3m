\section{Assessment}

\begin{frame}[t]{Assessment System}
\begin{itemize}
  \item In \textmark{summary}:

    \mode<presentation>{\vfill}
    \begin{itemize}
      \item \textgood{Final exam}: 40\% of final score.
        \begin{itemize}
          \item All contents included (theoretical, practical, and projects).
        \end{itemize}

      \mode<presentation>{\vfill}
      \item \textgood{Continuous evaluation}: 60\% of final score.
        \begin{itemize}
          \item \textmark{Tests}: 25\% of final score.
          \item \textmark{Projects}: 35\% of final score.
        \end{itemize}

    \mode<presentation>{\vfill}
    \item \textemph{Evaluation calls}:
      \begin{itemize}
        \item \textmark{Ordinary}: January.
        \item \textmark{Extraordinary}: June.
      \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Continuous evaluation}
\begin{itemize}
  \item Good continuous evaluation scores are \textemph{key} for success.

  \mode<presentation>{\vfill}
  \item \textgood{Elements}:
    \begin{itemize}
      \item \textmark{Evaluations during course}: 25\% of final score.
      \item \textmark{Projects and Labs}: 35\% of final score.
    \end{itemize}
  \vspace{1em}

  \mode<presentation>{\vfill}
  \item You \textbad{have not followed} continuous evaluation:
    \begin{itemize}
      \item if you get less than 2.0 in a project/lab, or,
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Ordinary round: Continuous evaluation}
\begin{itemize}
  \item If you follow the \textmark{continuous evaluation} process:
    \begin{itemize}
      \item \textgood{Final exam}: 40\%.
        \begin{itemize}
          \item Minimum required: \textbad{3.5}.
        \end{itemize}
      \item \textgood{Tests}: 25\%
        \begin{itemize}
          \item Minimum required: There is no minimum.
        \end{itemize}
      \item \textgood{Projects/Labs}: 35\%.
        \begin{itemize}
          \item Minimum required per project/lab: \textbad{2.0}.
        \end{itemize}
      \item If you do not achieve a minimum, the average is not computed and 
          \textbad{you have failed}.
    \end{itemize}

  \mode<presentation>{\vfill}
  \item \textemph{Bonus}:
    \begin{itemize}
      \item 1 point is added to your score if:
        \begin{itemize}
          \item You got at least 7.0 points in continuous evaluation, and,
          \item You got at least 6.0 points in the final exam.
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Evaluations during the course}
\begin{itemize}
  \item Tests carried out \textmark{during classes}.

  \mode<presentation>{\pause\vfill}
  \item Eeach test \textmark{may include}:
    \begin{itemize}
      \item Theoretical questions.
      \item Problem solving.
      \item \ldots
    \end{itemize}

  \mode<presentation>{\pause\vfill}
  \item Dates for each test are \textemph{already published}.

  \mode<presentation>{\pause\vfill}
  \item The average of the \textmark{best 4 tests} is taken.
    \begin{itemize}
      \item Not presented $\Rightarrow$ 0.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Labs}
\begin{itemize}
  \item Four lab sessions:
    \begin{itemize}
        \item Performed during lab sessions in groups of 2 students.
        \item Test or report after the session during a given period.
        \item \textgood{Weight} of each lab: 3\% of total mark.
        \item \textbad{Minumum mark}: 2 over 10.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Projects}
\begin{itemize}
  \item One project on performance oriented programming.
    \begin{itemize}
        \item Developed using C++ and OpenMP.
        \item Two deliveries.
        \begin{itemize}
          \item Week 8 (10/24): Sequential optimizations.
          \item Week 11 (11/14): Parallel optimizations.
        \end{itemize}

        \item Groups of 4 people.
        \item \textmark{Performance} is a key criterion.
        \item Marks include \textmark{quality of code}, \textmark{tests}, 
              and \textmark{performance evaluation}.
        \item Importance of \textmark{quality of project report}.
        \item \textgood{Weight}: 23\% of total mark.
          \begin{itemize}
            \item Sequential optimizations: 10\%.
            \item Parallel optimizations: 13\%.
          \end{itemize}
        \item \textbad{Minimum mark}: 2 over 10.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[t]{Ordinary round: NON-Continuous evaluation}
\begin{itemize}
  \item If you \textbad{did not follow} the \textgood{continuous evaluation} process:
    \begin{itemize}
      \item \textgood{Final exam} is worth \textbad{60\%} of final score.
      \item You need \textbad{8.33} in the \textgood{final exam} to pass the course.
    \end{itemize}

  \mode<presentation>{\vfill}
  \item \textbad{ADVICE}:
    \begin{itemize}
      \item Try hard to follow the continuous evaluation process.
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}[t]{Extraordinary round}
\begin{itemize}
  \item \textgood{Extraordinary exam} to be held in June.

  \mode<presentation>{\vfill}
  \item \textmark{Rules}:
    \begin{enumerate}
      \item The student \textgood{completed} the continuous evaluation:
        \begin{itemize}
          \item Extraordinary exam gives \textbad{40\%} 
                and continuous evaluation gives \textbad{60\%}.
          \item \textbad{Only applied} if the score in exam is at least \textbad{3.5}.
        \end{itemize}
      \item The student \textbad{did not complete} the continuous evaluation: 
        \begin{itemize}
          \item Final exam gives \textbad{100\%}.
        \end{itemize}
    \end{enumerate}

    \mode<presentation>{\vfill}
    \begin{itemize}
      \item For students having completed continuous evaluation 
            the \textgood{best of two options} is taken.
    \end{itemize}
\end{itemize}
\end{frame}
