\section{Pipelined access to cache}

\begin{frame}[t]{Pipelined access to cache}
\begin{itemize}
  \item \textmark{Goal}: Improve cache bandwidth. 

  \mode<presentation>{\vfill\pause}
  \item \textgood{Solution}: Pipelined access to the cache in multiple clock cycles.

  \mode<presentation>{\vfill\pause}
  \item \textgood{Effects}:
    \begin{itemize}
      \item Clock cycle can be shortened.
      \item A new access can be initiated every clock cycle.
      \item Cache bandwidth is increased.
      \item Latency is increased.
    \end{itemize}

  \mode<presentation>{\vfill\pause}
  \item Positive effect in \textmark{superscalar processors}.
\end{itemize}
\end{frame}
