\begin{itemize}
  \item \textgood{Modelo de comportamiento de memoria}:
    \begin{itemize}
      \item Las operaciones de memoria ocurren en \textgood{orden de programa}.
        \begin{itemize}
          \item Una lectura devuelve el valor de la última escritura en orden de programa.
        \end{itemize}
    \end{itemize}
  \mode<presentation>{\vfill\pause}
  \item Semántica definida por \textgood{orden de programa secuencial}:
    \begin{itemize}
      \item Razonamiento \textmark{simple} pero \textmark{restringido}.
        \begin{itemize}
          \item Resolver \textmark{dependencias de datos y control}.
        \end{itemize}
      \item Las operaciones \textmark{independientes} pueden ejecutarse en \textmark{paralelo}.
      \item Las optimizaciones \textmark{preservan la semántica}.
    \end{itemize}
\end{itemize}

\end{columns}
