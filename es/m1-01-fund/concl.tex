\section{Conclusión}

\begin{frame}[t]{Resumen}
\begin{itemize}
  \item Una secuencia de revoluciones:
    \begin{itemize}
      \item Microprocesadores $\rightarrow$ RISC con ILP y cachés $\rightarrow$ multi-cores.
    \end{itemize}

  \mode<presentation>{\pause\vfill}
  \item Nuevos modelos mejoran rendimiento (\bulletmark{DLP}, \bulletmark{TLP}, \bulletmark{RLP}), 
        pero requieren reestructurar las aplicaciones.

  \mode<presentation>{\pause\vfill}
  \item Diversidad en clases de computadores con variación en propiedades y requisitos.
    \begin{itemize}
      \item IoT/Empotrados, PMDs, Escritorio, Servidores, \emph{Cluster}/WSC.
    \end{itemize}

  \mode<presentation>{\pause\vfill}
  \item Nuevas arquitecturas que combinan \bulletmark{SIMD} y \bulletmark{MIMD}.

  \mode<presentation>{\pause\vfill}
  \item Diferencia entre \bulletmark{ISA} y \emph{arquitectura}.
\end{itemize}
\end{frame}
